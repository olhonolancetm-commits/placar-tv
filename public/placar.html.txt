    <script>
        // CONFIGURAÇÃO DO SEU FIREBASE
        const firebaseConfig = { 
            databaseURL: "https://placar-quadra-default-rtdb.firebaseio.com/" 
        };
        firebase.initializeApp(firebaseConfig);
        const database = firebase.database();

        // 🔄 ESCUTA AS ALTERAÇÕES DO BANCO EM TEMPO REAL
        database.ref('/placar_atual').on('value', (snapshot) => {
            const dadosPlacar = snapshot.val();
            
            if (dadosPlacar) {
                // Atualiza os nomes e trata os escudos para manter o tamanho fixo padrão
                document.getElementById('tvNomeCasa').innerText = dadosPlacar.nomeCasa || "MANDANTE";
                document.getElementById('tvEscudoCasa').src = dadosPlacar.urlEscudoCasa || "https://via.placeholder.com/250?text=CASA";
                
                document.getElementById('tvNomeVisitante').innerText = dadosPlacar.nomeVisitante || "VISITANTE";
                document.getElementById('tvEscudoVisitante').src = dadosPlacar.urlEscudoVisitante || "https://via.placeholder.com/250?text=VISITANTE";
                
                // Atualiza os Gols (subidos pelo ESP32 ou painel)
                document.getElementById('tvGolsCasa').innerText = dadosPlacar.golsCasa !== undefined ? dadosPlacar.golsCasa : 0;
                document.getElementById('tvGolsVisitante').innerText = dadosPlacar.golsVisitante !== undefined ? dadosPlacar.golsVisitante : 0;
                
                // Atualiza o Tempo de jogo
                document.getElementById('tempoTV').innerText = dadosPlacar.tempo || "60:00";
            }
        });
    </script>

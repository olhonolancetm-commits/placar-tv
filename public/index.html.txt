    <script>
        const firebaseConfig = { databaseURL: "https://placar-quadra-default-rtdb.firebaseio.com/" };
        firebase.initializeApp(firebaseConfig);
        const database = firebase.database();
        const auth = firebase.auth();
        let dadosUsuarioLogado = null;

        auth.onAuthStateChanged(user => {
            if (user) {
                database.ref('/usuarios/' + user.uid).once('value').then(snapshot => {
                    dadosUsuarioLogado = snapshot.val();
                    if(dadosUsuarioLogado) {
                        document.getElementById('userNome').innerText = dadosUsuarioLogado.responsavel;
                        document.getElementById('userTime').innerText = dadosUsuarioLogado.nomeTime;
                        mostrarAba('Agendamento');
                    }
                });
            } else { mostrarAba('Login'); }
        });

        function mostrarAba(aba) {
            document.getElementById('secaoLogin').style.display = aba === 'Login' ? 'block' : 'none';
            document.getElementById('secaoCadastro').style.display = aba === 'Cadastro' ? 'block' : 'none';
            document.getElementById('secaoAgendamento').style.display = aba === 'Agendamento' ? 'block' : 'none';
            document.getElementById('modalConfirmacao').style.display = 'none';
        }

        function atualizarOpcoesRival() {
            if(document.querySelector('input[name="financeiro"]:checked').value === "Rachar") {
                document.querySelector('input[name="tipoRival"][value="Procurar"]').checked = true;
                atualizarAbasRival();
            }
        }

        function atualizarAbasRival() {
            const tipoRival = document.querySelector('input[name="tipoRival"]:checked').value;
            if(document.querySelector('input[name="financeiro"]:checked').value === "Rachar" && tipoRival === "Conhecido") {
                alert("Para rachar o valor pelo site, selecione 'Procurar adversário no site'.");
                document.querySelector('input[name="tipoRival"][value="Procurar"]').checked = true;
                document.getElementById('campoRivalFixo').style.display = 'none';
                return;
            }
            document.getElementById('campoRivalFixo').style.display = tipoRival === 'Conhecido' ? 'block' : 'none';
        }

        function cadastrar() {
            const email = document.getElementById('cadEmail').value; const BlackPass = document.getElementById('cadSenha').value;
            const nome = document.getElementById('cadNome').value; const celular = document.getElementById('cadCelular').value;
            const time = document.getElementById('cadTime').value; const escudo = document.getElementById('cadEscudo').value;
            const ano = document.getElementById('cadAno').value; const sobre = document.getElementById('cadSobre').value;
            
            if(!email || !BlackPass || !nome || !celular || !time || !ano || !sobre) return alert("Preencha todos os campos!");

            auth.createUserWithEmailAndPassword(email, BlackPass).then(cred => {
                return database.ref('/usuarios/' + cred.user.uid).set({ 
                    responsavel: nome, celular: celular, nomeTime: time, urlEscudo: escudo, 
                    anoFundacao: ano, sobreTime: sobre, uid: cred.user.uid
                });
            }).catch(err => alert(err.message));
        }

        function logar() { auth.signInWithEmailAndPassword(document.getElementById('loginEmail').value, document.getElementById('loginSenha').value).catch(err => alert(err.message)); }
        function deslogar() { auth.signOut(); }

        function processarAgendamento() {
            const data = document.getElementById('dataJogo').value;
            const horario = document.getElementById('horarioJogo').value;
            const financeiro = document.querySelector('input[name="financeiro"]:checked').value;
            const tipoRival = document.querySelector('input[name="tipoRival"]:checked').value;

            if (!data || !horario) return alert("Escolha o dia e o horário!");
            document.getElementById('areaContatoRival').style.display = 'none';

            if (tipoRival === "Conhecido") {
                const rivalNome = document.getElementById('rivalNome').value || "Adversário";
                const rivalEscudo = document.getElementById('rivalEscudo').value || "https://via.placeholder.com/250?text=RIVAL";
                salvarNoFirebase(data, horario, dadosUsuarioLogado.nomeTime, dadosUsuarioLogado.urlEscudo, rivalNome, rivalEscudo, "Pendente", financeiro, dadosUsuarioLogado, null);
            } else {
                database.ref('/agendamentos').orderByChild('data').equalTo(data).once('value').then(snapshot => {
                    let matchID = null; let matchJogo = null;
                    snapshot.forEach(child => {
                        const jogo = child.val();
                        if (jogo.horario === horario && jogo.status === "Aguardando Rival" && jogo.modoFinanceiro === financeiro && jogo.nomeCasa !== dadosUsuarioLogado.nomeTime) {
                            matchID = child.key; matchJogo = child.val();
                        }
                    });

                    if (matchJogo) {
                        // Cruza os dados no agendamento e anexa os metadados do rival de quem estava esperando
                        database.ref('/agendamentos/' + matchID).update({
                            nomeVisitante: dadosUsuarioLogado.nomeTime,
                            urlEscudoVisitante: dadosUsuarioLogado.urlEscudo,
                            responsavelVisitante: dadosUsuarioLogado.responsavel,
                            celularVisitante: dadosUsuarioLogado.celular,
                            uidVisitante: dadosUsuarioLogado.uid, // Guarda o ID do rival para futuras consultas
                            status: "Pendente" 
                        }).then(() => {
                            // PUXA OS DADOS DO RIVAL (Dono do Time Casa que já estava registrado)
                            document.getElementById('modalTitulo').innerText = "💥 CONFRONTO ENCONTRADO!";
                            document.getElementById('modalTexto').innerText = "Jogo casado! Analise o perfil do seu adversário abaixo, combine os uniformes e chame a quadra para liberar!";
                            
                            // Preenche o Card de Perfil do Rival dinamicamente com as novas variáveis
                            document.getElementById('rivalPerfilEscudo').src = matchJogo.escudoCasaPerfil || matchJogo.urlEscudoCasa;
                            document.getElementById('rivalPerfilTime').innerText = matchJogo.nomeCasa;
                            document.getElementById('rivalPerfilAno').innerText = matchJogo.anoCasaPerfil || "Não Informado";
                            document.getElementById('rivalPerfilSobre').innerText = matchJogo.sobreCasaPerfil || "Sem descrição.";
                            document.getElementById('rivalRespNome').innerText = matchJogo.responsavelCasa;
                            document.getElementById('rivalRespCelular').innerText = matchJogo.celularCasa;
                            
                            document.getElementById('areaContatoRival').style.display = 'block';
                            document.getElementById('modalConfirmacao').style.display = 'block';
                        });
                    } else {
                        // Vai para a fila, levando todo o seu dossiê/perfil do time para o próximo ver
                        salvarNoFirebase(data, horario, dadosUsuarioLogado.nomeTime, dadosUsuarioLogado.urlEscudo, "Procurando...", "https://via.placeholder.com/250?text=?", "Aguardando Rival", financeiro, dadosUsuarioLogado, true);
                    }
                });
            }
        }

        function salvarNoFirebase(data, horario, casa, escudoCasa, visitante, escudoVisitante, status, financeiro, donoDados, colocarFila) {
            const jogoData = {
                data: data, horario: horario,
                nomeCasa: casa, urlEscudoCasa: escudoCasa,
                nomeVisitante: visitante, urlEscudoVisitante: escudoVisitante,
                responsavelCasa: donoDados.responsavel, celularCasa: donoDados.celular,
                uidCasa: donoDados.uid,
                // Leva o perfil do criador para o banco para o rival conseguir ler no match instantâneo
                anoCasaPerfil: donoDados.anoFundacao || "",
                sobreCasaPerfil: donoDados.sobreTime || "",
                escudoCasaPerfil: donoDados.urlEscudo || "",
                status: status, modoFinanceiro: financeiro
            };

            database.ref('/agendamentos').push().set(jogoData).then(() => {
                document.getElementById('modalTitulo').innerText = "Pedido Enviado!";
                if (colocarFila) {
                    document.getElementById('modalTexto').innerText = "Você está na lista de espera. O seu perfil de time foi publicado neste horário. Quando um rival aceitar, ele verá seus dados!";
                } else {
                    document.getElementById('modalTexto').innerText = "Agendamento pré-registrado. Clique abaixo para mandar mensagem para a administração!";
                }
                document.getElementById('modalConfirmacao').style.display = 'block';
            });
        }

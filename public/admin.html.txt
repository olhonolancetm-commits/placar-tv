    <script>
        // CONFIGURAÇÃO DO SEU FIREBASE
        const firebaseConfig = { 
            databaseURL: "https://placar-quadra-default-rtdb.firebaseio.com/" 
        };
        firebase.initializeApp(firebaseConfig);
        const database = firebase.database();

        // Usuário e Senha padrões do painel
        const USER_ADMIN = "admin";
        const PASSWORD_ADMIN = "123456";

        function fazerLogin() {
            const u = document.getElementById('usuario').value;
            const p = document.getElementById('senha').value;
            if (u === USER_ADMIN && p === PASSWORD_ADMIN) {
                document.getElementById('telaLogin').style.display = 'none';
                document.getElementById('painelAdmin').style.display = 'block';
                carregarAgendamentos();
            } else { 
                alert('Usuário ou senha inválidos!'); 
            }
        }
        
        function fazerLogout() { 
            document.getElementById('telaLogin').style.display = 'block'; 
            document.getElementById('painelAdmin').style.display = 'none'; 
        }

        // BUSCA TODOS OS AGENDAMENTOS E EXIBE OS DADOS DE CONTATO E FINANCEIRO
        function carregarAgendamentos() {
            database.ref('/agendamentos').on('value', (snapshot) => {
                const tabela = document.getElementById('tabelaAgendamentos');
                tabela.innerHTML = "";
                
                snapshot.forEach((childSnapshot) => {
                    const id = childSnapshot.key;
                    const jogo = childSnapshot.val();

                    // Tratamento de Status
                    let txtStatus = "Pendente";
                    let classeStatus = "status-pendente";
                    if(jogo.status === "Aguardando Rival") { 
                        txtStatus = "🔍 Buscando Rival"; 
                        classeStatus = "status-espera"; 
                    } else if(jogo.status === "Confirmado") { 
                        txtStatus = "✅ Confirmado"; 
                        classeStatus = "status-confirmado"; 
                    }

                    // Montagem inteligente dos botões de ações
                    let botoesAcao = "";
                    if(jogo.status === "Pendente") {
                        botoesAcao = `<button class="btn-confirmar" onclick="confirmarHorario('${id}')">Confirmar Horário</button>`;
                    } else if(jogo.status === "Confirmado") {
                        botoesAcao = `<button class="btn-placar" onclick="enviarParaTV('${jogo.nomeCasa}', '${jogo.urlEscudoCasa}', '${jogo.nomeVisitante}', '${jogo.urlEscudoVisitante}')">Enviar p/ TV 📺</button>`;
                    } else {
                        botoesAcao = `<span style="font-size:0.85rem; color:#888;">Aguardando Matchmaking...</span>`;
                    }

                    // Informações extras dos responsáveis para o ADM poder cobrar ou ligar
                    const infoCasa = `<span class="info-contato">Resp: ${jogo.responsavelCasa || 'Não informado'}<br>Tel: ${jogo.celularCasa || '-'}</span>`;
                    const infoVisitante = jogo.nomeVisitante !== "Procurando..." ? 
                        `<span class="info-contato">Resp: ${jogo.responsavelVisitante || 'Não informado'}<br>Tel: ${jogo.celularVisitante || '-'}</span>` : '';

                    const linha = `
                        <tr>
                            <td><strong>${jogo.data}</strong><br>${jogo.horario}</td>
                            <td><span class="badge-financeiro">${jogo.modoFinanceiro === 'Rachar' ? '💰 Racha (50%)' : '👑 Dono (100%)'}</span></td>
                            <td>
                                <img src="${jogo.urlEscudoCasa}" class="mini-escudo"><strong>${jogo.nomeCasa}</strong>
                                ${infoCasa}
                            </td>
                            <td>
                                <img src="${jogo.urlEscudoVisitante}" class="mini-escudo"><strong>${jogo.nomeVisitante}</strong>
                                ${infoVisitante}
                            </td>
                            <td class="${classeStatus}">${txtStatus}</td>
                            <td>${botoesAcao}</td>
                        </tr>
                    `;
                    tabela.innerHTML += linha;
                });
            });
        }

        // CONFIRMAÇÃO DO HORÁRIO (MUDA STATUS PARA CONFIRMADO)
        function confirmarHorario(id) {
            database.ref('/agendamentos/' + id).update({ status: "Confirmado" })
                .then(() => alert("Horário validado financeiramente e confirmado!"));
        }

        // ENVIA OS ESCUDOS E TIMES EM TEMPO REAL PARA O PLACAR DA TV
        function enviarParaTV(nomeCasa, escudoCasa, nomeVisitante, escudoVisitante) {
            database.ref('/placar_atual').update({
                nomeCasa: nomeCasa, 
                urlEscudoCasa: escudoCasa,
                nomeVisitante: nomeVisitante, 
                urlEscudoVisitante: escudoVisitante,
                golsCasa: 0, 
                golsVisitante: 0, 
                tempo: "60:00"
            }).then(() => alert("Sucesso! O jogo foi carregado no Placar da TV de 42\"."));
        }
